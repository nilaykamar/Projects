select b.*, a.avg from 
(select b.workid, avg(b.wordcount)
from (
        select workid, totalwords
        from work
        order by totalwords desc
        LIMIT 20) as a LEFT JOIN paragraph as b on a.workid = b.workid
group by b.workid
